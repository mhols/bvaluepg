\section{Motivation}

\begin{frame}{From Point Events to Counts}
\begin{center}
\maybeincludegraphics[width=0.72\textwidth]{figures/Figure_3.png}
\end{center}

\medskip
We observe point events, but the underlying spatial rate is unknown.
\end{frame}

\begin{frame}{Goal}
We aggregate events into grid cells:
\[
n_i = \#\{\text{events in cell } i\}.
\]

\medskip
We want to infer a spatially smooth intensity field
\[
\lambda_i > 0
\]
from noisy count observations.

\medskip
\begin{block}{Latent-field model}
\[
n_i \mid f_i \sim \Poisson(\lambda_i),
\qquad
\lambda_i = L(f_i),
\qquad
f\sim\N(\mu_0,\Sigma_0).
\]
\end{block}

The choice of link function \(L\) is central.
\end{frame}

\begin{frame}{Latent Gaussian Field}
We model spatial dependence through a Gaussian prior:
\[
f \sim \N(\mu_0,\Sigma_0).
\]

The covariance matrix is built from a spatial kernel:
\[
(\Sigma_0)_{ij}=K(x_i,x_j).
\]

\medskip
\begin{itemize}
\item \(f_i\): latent field value in cell \(i\)
\item \(K\): spatial covariance kernel
\item nearby cells have similar latent values
\end{itemize}
\end{frame}

\section{Why Replace the Sigmoid Link?}

\begin{frame}{Sigmoid Link}
The previous model used
\[
\lambda_i = \lambda\,\sigma(f_i),
\qquad
\sigma(f)=\frac{1}{1+e^{-f}}.
\]

\medskip
This guarantees positivity:
\[
\lambda_i >0.
\]

\medskip
But it also imposes a fixed upper bound:
\[
0<\lambda_i<\lambda.
\]
\end{frame}

\begin{frame}{Limitations of the Sigmoid Link}
The sigmoid link has three important limitations.

\medskip
\begin{enumerate}
\item \textbf{Extra scale parameter:}
      the maximum rate is fixed by \(\lambda\).

\medskip
\item \textbf{Fast saturation:}
      for moderately large \(f_i\),
      \[
      \sigma(f_i)\approx 1,
      \]
      so high and very high latent values become difficult to distinguish.

\medskip
\item \textbf{Slow posterior sampling:}
      P\'olya--Gamma augmentation gives a valid Gibbs sampler,
      but it becomes expensive on larger grids.
\end{enumerate}
\end{frame}

\begin{frame}{Sigmoid Saturation}
\begin{center}
\maybeincludegraphics[width=0.80\textwidth]{figures/link_functions.png}
\end{center}

\begin{alertblock}{Main issue}
The sigmoid changes very little after moderate positive values of \(f\).
The model can saturate even when the underlying rate should still increase.
\end{alertblock}
\end{frame}

\section{Soft-ramp / Softplus Link}

\begin{frame}{Soft-ramp Link}
We replace the bounded sigmoid link by the soft-ramp / softplus link:
\[
\lambda_i = \softplus(f_i),
\qquad
\softplus(f)=\log(1+e^f).
\]

\medskip
The link is positive, smooth, and unbounded.

\medskip
\begin{center}
\begin{tabular}{lll}
\toprule
Regime & Approximation & Interpretation \\
\midrule
\(f_i\ll 0\) & \(\softplus(f_i)\approx 0\) & low rate \\
\(f_i\gg 0\) & \(\softplus(f_i)\approx f_i\) & approximately linear growth \\
\bottomrule
\end{tabular}
\end{center}
\end{frame}



\begin{frame}{Observation Model}
After discretization into grid cells:
\[
n_i \mid f_i \sim \Poisson(\lambda_i),
\qquad
\lambda_i=\softplus(f_i).
\]

The likelihood contribution is
\[
p(n_i\mid f_i)
\propto
\softplus(f_i)^{n_i}
\exp\{-\softplus(f_i)\}.
\]

\medskip
\begin{alertblock}{Consequence}
The likelihood is non-Gaussian in \(f_i\), so the posterior is not available in closed form.
\end{alertblock}
\end{frame}



\section{Inference with the New Link}



\begin{frame}{Making the Softplus Link Tractable}
For each count value \(n\), approximate the one-dimensional likelihood by a Gaussian mixture in \(f\):
\[
p(n\mid f)
\approx
\sum_{k=1}^{K_n}
w_{n,k}\,
\N\!\left(f\mid \mu_{n,k},\sigma_n^2\right).
\]

\medskip
The mixture is precomputed once for each possible observed count.

\medskip
\begin{block}{Key effect}
The softplus likelihood becomes conditionally Gaussian.
\end{block}
\end{frame}

\begin{frame}{Example Likelihood Approximation}
\begin{center}
\maybeincludegraphics[width=0.76\textwidth]{figures/likelihood_mixture_residuals_n3_L1_sigma1.0.png}
\end{center}

\medskip
A precomputed Gaussian mixture approximates the Poisson--softplus likelihood for each observed count.
\end{frame}

\begin{frame}{Latent Mixture Variables}
Introduce a latent indicator for every cell:
\[
z_i \in \{1,\dots,K_{n_i}\}.
\]

Conditional on \(f_i\) and \(n_i\),
\[
p(z_i=k\mid f_i,n_i)
\propto
w_{n_i,k}\,
\N\!\left(f_i\mid \mu_{n_i,k},\sigma_{n_i}^2\right).
\]

\medskip
This categorical update is computed independently for all cells.
\end{frame}

\begin{frame}{Gibbs Sampler}
Repeat:

\medskip
\begin{enumerate}
\item Sample mixture assignments:
\[
z_i \sim p(z_i\mid f_i,n_i).
\]

\item Sample the full latent field:
\[
f\sim p(f\mid z,n).
\]
\end{enumerate}

\medskip
\begin{block}{Advantage}
Conditional on \(z\), the field update is Gaussian.
\end{block}
\end{frame}

\begin{frame}{Gaussian Conditional Posterior}
\small

Given \(z\), the approximate likelihood is Gaussian in \(f\):
\[
p(n,z\mid f)
\propto
\N(f\mid \mu_z,D),
\]
where
\[
\mu_z=
(\mu_{n_1,z_1},\dots,\mu_{n_N,z_N})^\top,
\qquad
D=\operatorname{diag}(\sigma_{n_1}^2,\dots,\sigma_{n_N}^2).
\]

Combining this with
\[
f\sim\N(\mu_0,\Sigma_0)
\]
gives
\[
f\mid z,n \sim \N(m_{\mathrm{post}},\Sigma_{\mathrm{post}}),
\]
with
\[
\Sigma_{\mathrm{post}}
=
(\Sigma_0^{-1}+D^{-1})^{-1},
\qquad
m_{\mathrm{post}}
=
\Sigma_{\mathrm{post}}
(\Sigma_0^{-1}\mu_0+D^{-1}\mu_z).
\]
\end{frame}

\begin{frame}{Why the Mixture Sampler is Fast}
The posterior covariance conditional on \(z\) is
\[
\Sigma_{\mathrm{post}}
=
(\Sigma_0^{-1}+D^{-1})^{-1}.
\]

Since \(D\) depends only on the observed counts \(n_i\),
\[
D \text{ is fixed during sampling.}
\]

\medskip
\begin{block}{Computational consequence}
The expensive matrix factorization is done once.
At each Gibbs step, only \(\mu_z\) and the posterior mean are updated.
\end{block}
\end{frame}
